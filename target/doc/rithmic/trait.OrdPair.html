Order a pair (2-tuple or 2-array)

use rithmic::OrdPair;

let (a, b) = (11, 3).ordered();
assert_eq!((a, b), (3, 11));

let [a, b] = [5, 17].ordered();
assert_eq!([a, b], [5, 17]);
