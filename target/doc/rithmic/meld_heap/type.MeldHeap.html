A priority queue

MeldHeap’s interface is similar to std::collections::BinaryHeap, but MeldHeap::meld runs in O(log n) time versus BinaryHeap::append’s O(n)

Examples

let q0 = MeldHeap::from_iter([2, 3, 5]);
let q1 = MeldHeap::from_iter([1, 4, 6]);
let mut q = q0.meld(q1);

assert_eq!(q.pop(), Some(6));
assert_eq!(q.pop(), Some(5));
assert_eq!(q.pop(), Some(4));
